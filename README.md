Library.IEEE;
USE IEEE.STD_LOGIC_1164.ALL;
ENTITY COUNT_UP IS
PORT(CLK : IN STD_LOGIC;
     RESET: IN STD_LOGIC;
    COUNTER : OUT STD_LOGIC_VECTURE(3 DOWNTO 0);
END ENTITY;
ARCHITECTURE BEHAVIOR OF COUNT_UP IS
SIGNAL COUNT_UP : STD_LOGIC_VECTURE(3 DOWNTO 0);
BEGIN
TIK : PROCESS(CLK)
       BEGIN
         IF RISING_EDGE OF (CLK) THEN
           IF RESET<= "0";
          COUNT_UP<= +"1";
         ELSE;
         RESET<= "1"
        COUNT_UP<="0"
         END IF;
        END IF;
     END PROCESS;
   COUNT_UP<=COUNTER;
END BEHAVIOR;
